### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-toggle-button
```

Then import the component in your component file.

```jsx
// ToggleButton.tsx
import * as ToggleButtonPrimitive from '@radix-ui/react-toggle-button';
```

### Basic example

```jsx live manual
// ToggleButton.tsx
// import * as ToggleButtonPrimitive from '@radix-ui/react-toggle-button';

const ToggleButton = styled(ToggleButtonPrimitive.Root, {
  appearance: 'none',
  backgroundColor: 'transparent',
  border: 'none',
  padding: '5px 10px',
  boxShadow: 'inset 0 0 0 1px gainsboro',
  overflow: 'hidden',
  borderRadius: 3,

  '&:focus': {
    outline: 'none',
    boxShadow: 'inset 0 0 0 1px dodgerblue, 0 0 0 1px dodgerblue',
  },

  '&[data-state=on]': {
    backgroundColor: 'gainsboro',
  },
});

render(<ToggleButton>Togggle</ToggleButton>);
```
