### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-scroll-area
```

Then import the component in your component file.

```jsx
// ScrollArea.tsx
import * as ScrollAreaPrimitive from '@interop-ui/react-scroll-area';
```

### Basic example

```jsx live manual
// ScrollArea.tsx
// import * as ScrollAreaPrimitive from '@interop-ui/react-scroll-area';
// import { CheckIcon } from '@modulz/radix-icons';

const ScrollArea = styled(ScrollAreaPrimitive.Root, {
  position: 'relative',
  // Root z-index set to 0 so we can set a new baseline for its children Apps may need to override
  // this if they have higher z-indices that conflict with their scrollbars, but they should not
  // need to change the z-indices for other elements in the tree.
  zIndex: 0,
  maxWidth: '100%',
  maxHeight: '100%',
  '&[data-dragging], &[data-scrolling]': {
    pointerEvents: 'auto !important',
  },
  '& [data-radix-scroll-area-position]::-webkit-scrollbar': {
    display: 'none',
  },
});

const Viewport = styled(ScrollAreaPrimitive.Viewport, {
  zIndex: 1,
  position: 'relative',
});

const ScrollbarY = styled(ScrollAreaPrimitive.ScrollbarY, {
  zIndex: 2,
  position: 'absolute',
  userSelect: 'none',
  transition: '300ms opacity ease',
  width: 8,
  right: 0,
  top: 0,
  bottom: 0,
});

const ScrollTrack = styled(ScrollAreaPrimitive.ScrollAreaTrack, {
  zIndex: -1,
  position: 'relative',
  width: '100%',
  height: '100%',
});

const ScrollThumb = styled(ScrollAreaPrimitive.ScrollAreaThumb, {
  backgroundColor: 'gainsboro',
  position: 'absolute',
  top: '0',
  left: '0',
  userSelect: 'none',
  borderRadius: 9999,
  willChange: `var(${ScrollAreaPrimitive.SCROLL_AREA_CSS_PROPS.scrollbarThumbWillChange})`,
  height: `var(${ScrollAreaPrimitive.SCROLL_AREA_CSS_PROPS.scrollbarThumbHeight})`,
  width: `var(${ScrollAreaPrimitive.SCROLL_AREA_CSS_PROPS.scrollbarThumbWidth})`,
});

render(
  <div style={{ height: 500 }}>
    <ScrollArea>
      <ScrollbarY>
        <ScrollTrack>
          <ScrollThumb />
        </ScrollTrack>
      </ScrollbarY>
      <Viewport>
        <div
          style={{
            height: '5000px',
            // width: '100%', expected this to work
            width: 200,
            backgroundImage:
              'linear-gradient(0deg, hsl(226,68%,56%) 0%, hsl(272,53%,50%) 50%, hsl(226,68%,56%) 100%)',
            backgroundSize: '100% 100px',
          }}
        />
      </Viewport>
    </ScrollArea>
  </div>
);
```

#### ScrollAreaViewport

`ScrollAreaViewport` is the viewport area of the `ScrollArea`.

#### ScrollAreaScrollbarX

`ScrollAreaScrollbarX` is the horizontal scrollbar.

#### ScrollAreaScrollbarY

`ScrollAreaScrollbarY` is the vertical scrollbar.

#### ScrollAreaButtonStart

`ScrollAreaButtonStart` is the start button in either `ScrollAreaScrollbarX` or `ScrollAreaScrollbarY`.

#### ScrollAreaButtonEnd

`ScrollAreaButtonEnd` is the end button in either `ScrollAreaScrollbarX` or `ScrollAreaScrollbarY`.

#### ScrollAreaTrack

`ScrollAreaTrack` is the track inside either `ScrollAreaScrollbarX` or `ScrollAreaScrollbarY`.

#### ScrollAreaThumb

`ScrollAreaThumb` is the thumb inside either `ScrollAreaScrollbarX` or `ScrollAreaScrollbarY`.

#### ScrollAreaCorner

`ScrollAreaCorner` is the corner where both vertical and horizontal scrollbars meet.
