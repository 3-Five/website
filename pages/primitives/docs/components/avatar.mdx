### Installation

Install the component from your command line.

```bash
npm install @radix-ui/react-avatar
```

Then import the component in your component file.

```jsx
// Avatar.tsx
import * as AvatarPrimitive from '@interop-ui/react-avatar';
```

### Basic example

```jsx live manual
// Avatar.tsx
// import * as AvatarPrimitive from '@interop-ui/react-avatar';

const Avatar = styled(AvatarPrimitive, {
  display: 'inline-flex',
  alignItems: 'center',
  justifyContent: 'center',
  verticalAlign: 'middle',
  overflow: 'hidden',
  userSelect: 'none',
  width: 48,
  height: 48,
  borderRadius: 24,
});

const AvatarImage = styled(AvatarPrimitive.Image, {
  width: '100%',
  height: '100%',
  objectFit: 'cover',
});

const AvatarFallback = styled(AvatarPrimitive.Fallback, {
  width: '100%',
  height: '100%',
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  backgroundColor: 'dodgerblue',
});

render(
  <Avatar>
    <AvatarImage src="https://picsum.photos/id/1005/400/400" />
    <AvatarFallback>UI</AvatarFallback>
  </Avatar>
);
```

#### Grouped

```jsx live manual compact
// Avatar.tsx
// import * as AvatarPrimitive from '@interop-ui/react-avatar';

  display: 'inline-flex',

  boxShadow: '0 0 0 1px white',

    marginLeft: -12,

  width: '100%',
  height: '100%',
  objectFit: 'cover',
});

  width: '100%',
  height: '100%',
  display: 'flex',
  alignItems: 'center',
  justifyContent: 'center',
  backgroundColor: 'dodgerblue',
});

render(
  <>
    <Avatar>
      <AvatarImage src="https://picsum.photos/id/1006/400/400" />
      <AvatarFallback>AB</AvatarFallback>
    </Avatar>
    <Avatar>
      <AvatarImage src="https://picsum.photos/id/1001/400/400" />
      <AvatarFallback>CD</AvatarFallback>
    </Avatar>
    <Avatar>
      <AvatarImage src="https://picsum.photos/id/1008/400/400" />

    </Avatar>
    <Avatar>
      <AvatarFallback>+10</AvatarFallback>
    </Avatar>
  </>
);
```
